# базовое определение
[[определение argc  и argv]]

# подробности
argc и argv - это аргументы главной функции(main) которые дают информацию о аргументах командной строки, переданой текущей прогграме.


- значением argc является число командых аргументов
- значением argv является вектор значений командных аргументов

argc всегда имеет значение как минимум 1, потому что имя прогграмы является командым аргументом
argv всегда имеет хотя бы 1 значение в массиве(имя прогграмы)

именно поэтому отсчет argc рекомендуется начинать с 2, а индексом argv начинать 1 


# примеры
минимальная версия echo:
```c
#include<stdio.h>

int main(int argc,char *argv[])
{
for(int i=1;i<argc;++i)
{
puts(argv[i]);
}
}
```

узнать, определен ли макрос:
```c
#include<stdio.h>


int main(int argc,char *argv[])
{

for(int i=1;i<argc;++i)
{
#ifdef argv[i]
printf("макрос %s определен",argv[i]);
return;
#endif

#ifndef argv[i]
printf("макрос %s  не определен",argv[i]);
return;
#endif
}

}
